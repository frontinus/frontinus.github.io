705HAL_StatusTypeDef EEPROM_Write(I2C_HandleTypeDef *hi2c, uint16_t memAddr, uint8_t *pData, uint16_t len) {
706 HAL_StatusTypeDef status = HAL_OK;
707 uint16_t bytesWritten = 0;
708
709 while (bytesWritten < len) {
710 // Calculate how many bytes can be written in the current page
711 uint16_t bytesToWriteInPage;
712 uint16_t currentPageStartAddr = memAddr & (~(EEPROM_PAGE_SIZE - 1)); // Calculate start address of the current page
713 uint16_t bytesLeftInPage = EEPROM_PAGE_SIZE - (memAddr - currentPageStartAddr);
714
715 bytesToWriteInPage = (len - bytesWritten < bytesLeftInPage) ? (len - bytesWritten) : bytesLeftInPage;
716
717 // Check if device is ready before writing
718 // Important: Wait extra long here in case the previous write cycle is still finishing
719 status = HAL_I2C_IsDeviceReady(hi2c, EEPROM_I2C_ADDR, 5, EEPROM_WRITE_TIME * 2);
720 if (status != HAL_OK) {
721 return status; // Device not ready, even after waiting
722 }
723
724 // Perform the memory write for the current chunk (up to a page boundary)
725 status = HAL_I2C_Mem_Write(hi2c, EEPROM_I2C_ADDR, memAddr, I2C_MEMADD_SIZE_16BIT, pData + bytesWritten, bytesToWriteInPage, I2C_TIMEOUT);
726
727 if (status != HAL_OK) {
728 return status; // Write failed
729 }
730 HAL_Delay(EEPROM_WRITE_TIME);
731
732 bytesWritten += bytesToWriteInPage;
733 memAddr += bytesToWriteInPage;
734 }
735
736 return status;
737} // end of Data_Storage
