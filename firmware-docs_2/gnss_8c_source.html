332char calc_checksum(char* buffer) {
333 unsigned char res = 0; // 0 ^ A = A
334
335 for(int i = 0; buffer[i] != '\0'; i++) {
336 if(buffer[i] == '$') {
337 res = 0;
338 continue;
339 } else if(buffer[i] == '*')
340 break;
341
342 res ^= buffer[i];
343 }
344
345 return res;
346}

403int check_nmea(char* buffer, size_t max_size, const NMEA_CMD* cmd) {
404 if(buffer[0] != '$')
405 return NMEA_INVALID; // Not an NMEA sentence
406
407 if(cmd != NULL) {
408 if(strncmp(buffer + 1, cmd->name, strlen(cmd->name)) != 0)
409 return NMEA_UNEXPECTED; // Not the expected command
410
411 // Find end of the sentence
412 // Points at the \n character
413 char* end = memchr(buffer+1, '\n', max_size);
414
415 // end - buffer gives the length of the string.
416 // strlen(cmd->name) gets optimized out by the compiler because
417 // it is a const char*, aka string literal.
418 // must have at minimum 6 characters + name: %name*FF\r\n
419 if(end == NULL){
420 return NMEA_INVALID;
421 }
422
423 if((end - buffer) < strlen(cmd->name)+6)
424 return NMEA_INVALID; // Not enough data to check checksum
425
426 long checksum = strtol(&end[-3], NULL, 16); // Provided
427 long calc = calc_checksum(buffer); // Calculated
428
429 if(checksum != calc)
430 return NMEA_CHECKSUM; // Checksum mismatch
431 }
432
433 return NMEA_VALID;
434}
