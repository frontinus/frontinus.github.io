14uint32_t servo_init(servo_t *servo, uint32_t pwm_freq, uint32_t target_arr) {
15 TIM_MasterConfigTypeDef master_conf = {0};
16// TIM_ClockConfigTypeDef sClockSourceConfig = {0};
17 TIM_OC_InitTypeDef oc_conf = {0};
18
19 // see __HAL_TIM_SET_COMPARE source
20 switch(servo->pwm_channel) {
21 case TIM_CHANNEL_1:
22 servo->TIM_CCR = &(servo->htim->Instance->CCR1);
23 break;
24 case TIM_CHANNEL_2:
25 servo->TIM_CCR = &(servo->htim->Instance->CCR2);
26 break;
27 case TIM_CHANNEL_3:
28 servo->TIM_CCR = &(servo->htim->Instance->CCR3);
29 break;
30 default:
31 servo->TIM_CCR = &(servo->htim->Instance->CCR4);
32 break;
33 }
34
35 servo->psc = (uint32_t) servo->bus_freq / (pwm_freq * target_arr);
36 servo->arr = (uint32_t) servo->bus_freq / (pwm_freq * (servo->psc + 1)) - 1;
37
38 uint32_t timer_freq = servo->bus_freq / (servo->psc + 1);
39
40 // /1000 because periods are in ms
41 servo -> minpos_ccr = (uint32_t) (servo->minpos_period_ms * timer_freq) / 1000;
42 servo -> maxpos_ccr = (uint32_t) (servo->maxpos_period_ms * timer_freq) / 1000;

44
45 // Timer deinit
46 HAL_TIM_PWM_DeInit(servo->htim);

55 servo->htim->Init.Prescaler = servo->psc;
56 servo->htim->Init.CounterMode = TIM_COUNTERMODE_UP;

58 servo->htim->Init.Period = servo->arr;
59
60 servo->htim->Init.ClockDivision = TIM_CLOCKDIVISION_DIV1;

70

72 master_conf.MasterOutputTrigger = TIM_TRGO_RESET;
73 master_conf.MasterSlaveMode = TIM_MASTERSLAVEMODE_DISABLE;

103float servo_moveto_deg(servo_t *servo, float angle) {
104 float o_angle = angle;
105
106 if (angle < servo->min_angle) {
107 angle = servo->min_angle;
108 errno = ERANGE;
109 }
110
111 if (angle > servo->max_angle) {
112 angle = servo->max_angle;
113 errno = ERANGE;
114 }
115
116 // angle to ccr
117 uint32_t ccr = servo_ccr(servo, servo_dutycycle(servo, angle));
118 servo_update_ccr(servo, ccr);
119
120 return angle - o_angle; // 0 on success
121}

123float servo_rawmove(servo_t *servo, uint32_t ccr) {
124 if (ccr <= servo->maxpos_ccr && ccr >= servo->minpos_ccr) {
125 servo_update_ccr(servo, ccr);
126
127 // ccr to angle
128 return servo_angle(servo, servo_dutycycle_ccr(servo, ccr));
129 } else {
130 errno = ERANGE;
131 return 0;
132 }
133}

135float servo_dutycycle(servo_t *servo, float angle) {
136 return (angle - servo->min_angle) / (servo->max_angle - servo->min_angle);
137}

139float servo_dutycycle_ccr(servo_t *servo, uint32_t ccr) {
140 return ((float) (ccr - servo->minpos_ccr)) / ((float) (servo->maxpos_ccr - servo->minpos_ccr));
141}

143uint32_t servo_ccr(servo_t *servo, float dutycycle) {
144 return (dutycycle * (servo->maxpos_ccr - servo->minpos_ccr)) + servo->minpos_ccr;
145}

147float servo_angle(servo_t *servo, float dutycycle) {
148 return (dutycycle * (servo->max_angle - servo->min_angle)) + servo->min_angle;
149}
